from neuroio.base import APIBase, APIBaseAsync, APIBaseBase

class SourcesBase(APIBaseBase):

    def get_url(self, key: str = None) -> str:

        if key:

            return self.base_url + f"/v1/sources/{key}/"

        else:

            return self.base_url + "/v1/sources/"
